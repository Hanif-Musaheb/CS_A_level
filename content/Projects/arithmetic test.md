        e=12

        a=c*random.randint(2,12)
    
    else :
        a=random.randint(1,e)
        c=random.randint(1,e)
        
    b=equation_type[rand_num]

    
    user_input=int(input('{}{}{}= '.format(a,b,c)))

    else:
        wrong+=1
        print('wrong')
